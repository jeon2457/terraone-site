(Storage)

공개전환일 경우(테스트중일때...) ===>

rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    match /{allPaths=**} {
      allow read: if true;
      allow write: if true;
    }
  }
}


보안전환일 경우(실제 운영중일때...) ===>

rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    match /{allPaths=**} {
      allow read: if true;  // 누구나 읽기 가능 (이미지 보기용)
      allow write: if request.auth != null;  // 로그인한 사용자만 쓰기
    }
  }
}
